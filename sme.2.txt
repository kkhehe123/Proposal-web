            function renderCampaigns() {
                campaignGrid.innerHTML = '';
                const filteredCampaigns = campaignData.filter(campaign => {
                    const goalMatch = currentGoalFilter === 'all' || campaign.goal === currentGoalFilter;
                    const partnerMatch = currentPartnerFilter === 'all' || campaign.partner === currentPartnerFilter;
                    return goalMatch && partnerMatch;
                });

                if (filteredCampaigns.length === 0) {
                    campaignGrid.innerHTML = `<div class="col-span-1 md:col-span-2 lg:col-span-3 text-center text-gray-500 py-10">

                    return;

                    filterGroup.querySelector('.active').classList.remove('active');
                    event.target.classList.add('active');

                    if (filterGroup.id === 'goal-filters') {
                        currentGoalFilter = filterValue;
                    } else if (filterGroup.id === 'partner-filters') {
                        currentPartnerFilter = filterValue;
                    }
                    
                    renderCampaigns();
                }
            }

            partnerFilters.addEventListener('click', (e) => handleFilterClick(partnerFilters, e));
